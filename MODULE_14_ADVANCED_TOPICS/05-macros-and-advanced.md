## Property Wrappers

Property wrappers encapsulate common property patterns:

```swift
// Custom property wrapper
@propertyWrapper
struct Clamped<T: Comparable> {
    private var value: T
    private let range: ClosedRange<T>
    
    init(wrappedValue: T, _ range: ClosedRange<T>) {
        self.range = range
        self.value = min(max(wrappedValue, range.lowerBound), range.upperBound)
    }
    
    var wrappedValue: T {
        get { value }
        set { value = min(max(newValue, range.lowerBound), range.upperBound) }
    }
}

// Usage
struct AudioPlayer {
    @Clamped(0...100) var volume: Int = 50
    @Clamped(0.5...2.0) var playbackRate: Double = 1.0
}

var player = AudioPlayer()
player.volume = 150  // Clamped to 100
player.volume = -10  // Clamped to 0

// Built-in property wrappers you use every day:
// @State, @Binding, @Published, @AppStorage, @FocusState, @Environment
```

### Property Wrapper with `projectedValue`

```swift
@propertyWrapper
struct Validated<T> {
    private var value: T
    private var isValid: Bool = false
    
    var wrappedValue: T {
        get { value }
        set {
            value = newValue
            isValid = validate(newValue)
        }
    }
    
    // Accessed via $property
    var projectedValue: Bool { isValid }
    
    private func validate(_ value: T) -> Bool {
        // Validation logic
        return true
    }
}

struct LoginForm {
    @Validated var email: String = ""
}

let form = LoginForm()
let isEmailValid = form.$email  // Bool from projectedValue
```

## Swift Macros (Swift 5.9+)

Macros generate code at compile time using Swift syntax.

### Types of Macros

```swift
// Freestanding — used standalone
#stringify(1 + 2)  // "#stringify" → produces "1 + 2" and 3

// Attached — applied to declarations
@Observable class Store { }
@Model final class User { }
```

### How @Observable Works

```swift
// What you write
@Observable
class UserStore {
    var name: String = ""
    var count: Int = 0
}

// What the macro generates (simplified)
class UserStore {
    var name: String = "" {
        get { _$observationRegistrar.access(self, keyPath: \.name); return _name }
        set { _$observationRegistrar.withMutation(self, keyPath: \.name) { _name = newValue } }
    }
    var count: Int = 0 { /* similar observation code */ }
    
    internal let _$observationRegistrar = Observation.ObservationRegistrar()
    
    internal nonisolated func access<Member>(
        keyPath: KeyPath<UserStore, Member>
    ) { _$observationRegistrar.access(self, keyPath: keyPath) }
    
    internal nonisolated func withMutation<Member, T>(
        keyPath: KeyPath<UserStore, Member>,
        _ mutation: () throws -> T
    ) rethrows -> T {
        try _$observationRegistrar.withMutation(self, keyPath: keyPath, mutation)
    }
}
```

### Creating a Custom Macro

```swift
// Package.swift — declare macro
.macro(
    name: "Logging",
    dependencies: ["LoggingMacros"]
)

// Declaration
@attached(member, names: named(log))
public macro Logging() = #externalMacro(module: "LoggingMacros", type: "LoggingMacro")

// Implementation (in separate module)
import SwiftSyntax
import SwiftSyntaxMacros

public struct LoggingMacro: MemberMacro {
    public static func expansion(
        of node: AttributeSyntax,
        providingMembersOf declaration: some DeclGroupSyntax,
        in context: some MacroExpansionContext
    ) throws -> [DeclSyntax] {
        return ["""
            func log(_ message: String) {
                print("[\\(type(of: self))] \\(message)")
            }
            """]
    }
}

// Usage
@Logging
struct Service {
    func doWork() {
        log("Starting work")  // Generated by macro
    }
}
```

## Result Builders

Power SwiftUI's DSL syntax:

```swift
// Simplified SwiftUI-like result builder
@resultBuilder
struct ViewBuilder {
    static func buildBlock(_ components: View...) -> [View] {
        components
    }
    
    static func buildOptional(_ component: [View]?) -> [View] {
        component ?? []
    }
    
    static func buildEither(first component: [View]) -> [View] { component }
    static func buildEither(second component: [View]) -> [View] { component }
}

// Usage — creates DSL
func makeView(@ViewBuilder content: () -> [View]) -> ContainerView {
    ContainerView(children: content())
}

makeView {
    Label("Title")
    if isLoggedIn {
        Button("Profile")
    } else {
        Button("Login")
    }
}
```

## Advanced Generics

```swift
// Generic constraint with multiple protocols
func processCollection<T: Collection>(_ collection: T) where T.Element: Hashable & Codable {
    let set = Set(collection)
    let encoded = try? JSONEncoder().encode(Array(set))
}

// Primary associated types (Swift 5.7+)
protocol Sequence<Element> { ... }
func process<S: Sequence<Int>>(_ seq: S) { }  // Shorthand

// Generic where clauses on extensions
extension Collection where Element: Numeric {
    var sum: Element { reduce(0, +) }
}

// Conditional conformance
extension Optional: Hashable where Wrapped: Hashable { }

// Opaque result type with constraint
func makeStorage<T: Codable>() -> some StorageProtocol where StorageProtocol.Value == T {
    ConcreteStorage<T>()
}
```

## @dynamicMemberLookup

```swift
@dynamicMemberLookup
struct JSON {
    private var dict: [String: Any]
    
    init(_ dict: [String: Any]) { self.dict = dict }
    
    subscript(dynamicMember key: String) -> JSON {
        JSON((dict[key] as? [String: Any]) ?? [:])
    }
    
    var string: String? { dict.values.first as? String }
    var int: Int? { dict.values.first as? Int }
}

let json = JSON(["user": ["name": "Alice", "age": 30]])
let name = json.user.name.string  // "Alice" — dot syntax without explicit subscript
```

## Interview Q&A

**Q: What are Swift macros and how do they differ from generics?**  
A: Macros generate code at compile time by transforming Swift syntax trees. They reduce boilerplate by writing code that you'd otherwise write manually. Generics are a type system feature for reusable algorithms over multiple types. Macros are code generation; generics are type abstraction.

**Q: How does @Observable differ from ObservableObject?**  
A: `@Observable` (iOS 17) uses a macro to generate fine-grained per-property observation tracking. SwiftUI tracks exactly which properties each view reads and only re-renders when those specific properties change. `ObservableObject` fires `objectWillChange` for any `@Published` property change, causing all observers to re-render regardless of which property changed.

**Q: What is a Result Builder?**  
A: `@resultBuilder` is an attribute that lets you define DSL-like syntax where multiple expressions in a closure are combined into a single value. SwiftUI's `@ViewBuilder` is the most prominent example — it combines multiple view expressions into a `TupleView`. HTML builders, test builders, and query builders use the same pattern.

## Quick Revision

- Property wrapper: `wrappedValue` (main value) + `projectedValue` ($prefix)
- `@Clamped`, `@Validated` — common custom wrapper examples
- Swift macros (5.9+): compile-time code generation via syntax transformation
- `@Observable`: generates observation tracking for fine-grained view updates
- `@attached(member)`: adds members; `@attached(conformance)`: adds protocol conformances
- Result builders: DSL syntax — `@ViewBuilder` in SwiftUI
- `@dynamicMemberLookup`: enables dot-syntax subscript access
- Custom macros live in separate module, use SwiftSyntax
